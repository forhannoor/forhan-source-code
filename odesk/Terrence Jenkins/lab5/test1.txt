//This is a code sample from a C++ program.

Student::Student(void)
{
	name = 0;
	major = 0;
	creditHours = 0;
	gpa = 0;
	numStudents++;
}

//Explicit constructor
Student::Student(char* n)
{
	name = new char[strlen(n) + 1];
	strcpy(name, n);
	major = 0;
	creditHours = 0;
	gpa = 0;
	numStudents++;
}

//Copy Constructor
Student::Student(Student &original)
{
	name = new char[strlen(original.name) + 1];
	strcpy(name, original.name);
	major = new char[strlen(original.major) + 1];
	strcpy(major, original.major);
	creditHours = original.creditHours;
	gpa = original.gpa;
	numStudents++;
}

//Function to allow the user to set the name of the student
void Student::setName(char* n)
{
	name = new char[strlen(n) + 1];
	strcpy(name, n);
}

//Function to allow the user to set the major
void Student::setMajor(char* m)
{
	major = new char[strlen(m) + 1];
	strcpy(major, m);
}

//Function to allow the user to set the number of credit hours the student has completed
void Student::setCreditHours(int credH)
{
	//Check to make sure that the value passed makes sense
	if(credH >= 0)
		creditHours = credH;
	else
	{
		cout << "Invalid number of credit hours.  Exiting program" << endl;
		exit(EXIT_FAILURE);
	}
}
